Delay <- FD$Delay
sims <- 10^4 - 1
ans <- numeric(sims)
for (i in 1:sims) {
  index <- sample(4029, 2906, FALSE)
  ans[i] <- mean(Delay[index]) - mean(Delay[-index])
}

OBS <- tapply(Delay, FD$Carrier, mean)
OBS_DIFF <- OBS[1] - OBS[2]

pval <- (sum(ans < OBS_DIFF && ans > -OBS_DIFF) + 1)/(sims + 1)
